%---------------- CUSTOM CLASS LOGIC ----------------%

\def \name#1{\def\@name{#1}}
\def \@name {}

\def \addressSep {$\diamond$}

\let \@addressone \relax
\let \@addresstwo \relax
\let \@addressthree \relax

\def \address #1{
  \@ifundefined{@addresstwo}{
    \def \@addresstwo {#1}
  }{
  \@ifundefined{@addressthree}{
  \def \@addressthree {#1}
  }{
     \def \@addressone {#1}
  }}
}

\def \printaddress #1{
  \begingroup
    \def \\ {\addressSep\ }
    \centerline{\small #1}
  \endgroup
  \par
  \vspace{1pt}
}

\def \printname {
  \begingroup
    \hfil{\MakeUppercase{\LARGE\bfseries \@name}}\hfil
    \vspace{2pt}\break
  \endgroup
}

\AtBeginDocument{
  \printname
  \@ifundefined{@addressone}{}{\printaddress{\@addressone}}
  \@ifundefined{@addresstwo}{}{\printaddress{\@addresstwo}}
  \@ifundefined{@addressthree}{}{\printaddress{\@addressthree}}
  \vspace{-4pt}
}

\newenvironment{rSection}[1]{
  \vspace{6pt}
  {\MakeUppercase{\bfseries #1}}
  \vspace{2pt}
  \hrule height 0.8pt
  \vspace{-2pt}
  \begin{list}{}{
    \setlength{\leftmargin}{1.5em}
  }
  \item[]
}{
  \end{list}
  \vspace{-6pt}
}

\newenvironment{rSubsection}[4]{
 {\bfseries #1} \hfill {#2}
 \ifthenelse{\equal{#3}{}}{}{
  \\
  {\em #3} \hfill {\em #4}
  }
  \vspace{-4pt}
  \begin{list}{$\cdot$}{\leftmargin=0em}
   \itemsep -0.5em \vspace{-0.5em}
  }{
  \end{list}
  \vspace{-2pt}
}
